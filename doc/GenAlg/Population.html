def crossover!
  (size * crossover_rate).to_i.times do
    new_chromosomes = random_chromosome.crossover(random_chromosome)
    add_chromosomes(new_chromosomes)
  end

def selection!
  @chromosomes.sort! do |a, b|
    b.fitness <=> a.fitness
  end
  @chromosomes.slice!(max_size .. size) \
    if size > max_size
end
